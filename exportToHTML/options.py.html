17       pygame.init()

25           #increase and decrease buttons for screen size, (change screen size by 1 times the number displayed with variable)
26           smallerScreenButton,biggerScreenButton=display.adjustableNum("Screen Size", str(settings.width),settings.height*0.5,(0,0,0))
27           smallerScreenButtonx10 = display.button("<<", settings.width * 0.1, settings.width * 0.25, settings.width * 0.27,settings.height * 0.5, (0, 0, 0))
28           smallerScreenButtonx100 = display.button("<<<", settings.width * 0.1, settings.width * 0.25, settings.width * 0.165,settings.height * 0.5, (0, 0, 0))
29           biggerScreenButtonx10 = display.button(">>", settings.width * 0.1, settings.width * 0.25,settings.width * 0.7275, settings.height * 0.5, (0, 0, 0))
30           biggerScreenButtonx100 = display.button(">>>", settings.width * 0.1, settings.width * 0.25,settings.width * 0.83, settings.height * 0.5, (0, 0, 0))
31   
32           for event in pygame.event.get():
33               if event.type == pygame.MOUSEBUTTONDOWN:
34                   #exit button
35                   if exitButton.collidepoint(event.pos):
36                       return
37                   #shrink screen size
38                   elif smallerScreenButton.collidepoint(event.pos):
39                       settings.width -= 1
40                       settings.changeScreenSize()
41                   #shrink screen size by 10
42                   elif smallerScreenButtonx10.collidepoint(event.pos):
43                       settings.width-=10
44                       settings.changeScreenSize()
45                   #shrink screen size by 100
46                   elif smallerScreenButtonx100.collidepoint(event.pos):
47                       settings.width -= 100
48                       settings.changeScreenSize()
49                   #increase screen size
50                   elif biggerScreenButton.collidepoint(event.pos):
51                       settings.width+=1
52                       settings.changeScreenSize()
53                   #increase screen size by 10
54                   elif biggerScreenButtonx10.collidepoint(event.pos):
55                       settings.width+=10
56                       settings.changeScreenSize()
57   
58                   #increase screen size by 100
59                   elif biggerScreenButtonx100.collidepoint(event.pos):
60                       settings.width+=100
61                       settings.changeScreenSize()
62               #if user inputs o then show admin options
63               if event.type==pygame.KEYDOWN:
64                   if event.key == pygame.K_o:
65                       settingAlterMenu()
66   
67           pygame.display.flip()
