3    import game
4    import settings
5    import display
6    import pygame
7    import options
8    
9    # boolean for determining to if the user is quitting
10   quitting = False
11   
12   # initiate pygame
13   pygame.init()
14   
15   #make the main menu image the correct size
16   display.image_resize(f"MainMenuBackground.png", settings.width * 2, settings.height)
17   
18   #loop
19   while True:
20   
21       #main menu background
22       display.image(f"newMainMenuBackground.png", 0, 0)
23       #button to start game
24       startButton = display.button("START", settings.width * 0.3225, settings.width * 0.25, settings.width * 0.5,settings.height * 0.375, (0, 0, 0))
25       #button to show options
26       optionButton = display.button("OPTIONS", settings.width * 0.3225, settings.width * 0.25, settings.width * 0.5, settings.height * 0.6, (0, 0, 0))
27       #button to quit the game
28       quitButton = display.button("QUIT", settings.width * 0.3225, settings.width * 0.25, settings.width * 0.5,settings.height * 0.825, (0, 0, 0))
29   
30   
31       for event in pygame.event.get():
32           if event.type == pygame.MOUSEBUTTONDOWN:
33               #start button clicked start game
34               if startButton.collidepoint(event.pos):
35                   game.startGame()
36               #option button clicked display option menu
37               elif optionButton.collidepoint(event.pos):
38                   options.optionsMenu()
39               #tell program that the user is quitting if quit button clicked
40               elif quitButton.collidepoint(event.pos):
41                   quitting = True
42       #if quitting exit loop
43       if quitting:
44           break
45   
46       pygame.display.flip()
