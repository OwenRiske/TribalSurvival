3    import pygame
4    
5    import toolBox
6    import display
7    import settings
8    import disaster
9    
10   
11   def startGame():
12       turn=1
13   
14       #starting resources
15       peopleAmount=0
16       coconutAmount=2
17       boatAmount=0
18       medicineAmount=0
19       blanketAmount=0
20       activeSwordAmount=0
21       unactiveSwordAmount=0
22       treeAmount=1
23       superTreeAmount=0
24       spearAmount=0
25       netAmount=0
26   
27       #makes it so the player gets a random resource
28       peopleAmount,coconutAmount,boatAmount,blanketAmount,medicineAmount,activeSwordAmount, treeAmount, superTreeAmount, spearAmount, netAmount=toolBox.buyPerson(peopleAmount,coconutAmount,boatAmount,blanketAmount,medicineAmount,activeSwordAmount, treeAmount, superTreeAmount, spearAmount, netAmount)
29   
30       #array for the disaster that the user will get
31       disasters=[]
32   
33       #keep track of turns
34       turn=1
35   
36       pygame.init()
37   
38   
39       while True:
40           # set background
41           display.background("newbackground.png")
42   
43   
44           #button for the game menu
45           #button for getting people
46           buyPerson = display.button("Buy Person", settings.width*0.3225, settings.width // 4,settings.width*0.17, settings.height*0.885, (0, 0, 0))
47           #button for trade
48           tradeButton= display.button("Trade", settings.width*0.3225, settings.width // 4,settings.width*0.5, settings.height * 0.125, (0, 0, 0))
49           #end turn button
50           doneTurn = display.button("Done", settings.width *0.3225, settings.width // 4, settings.width*0.5,settings.height *0.885, (0, 0, 0))
51           #quit game button
52           quitButton=display.button("Quit",settings.width*0.3225,settings.width*0.25,settings.width*0.83,settings.height*0.885,(0,0,0))
53           # display button for the resource window
54           resourceButton = display.button("Check Resources", settings.width * 0.3225, settings.width // 5,settings.width * 0.83, settings.height * 0.125, (0, 0, 0))
55   
56           #display the coconut amount and people amount
57           toolBox.peopleAndCoconuts(peopleAmount, coconutAmount)
58   
59   
60   
61   
62           for event in pygame.event.get():
63               if event.type == pygame.MOUSEBUTTONDOWN:
64                   #acquire person
65                   if buyPerson.collidepoint(event.pos):
66                       peopleAmount,coconutAmount,boatAmount,blanketAmount,medicineAmount,activeSwordAmount, treeAmount, superTreeAmount, spearAmount, netAmount=toolBox.buyPerson(peopleAmount,coconutAmount,boatAmount,blanketAmount,medicineAmount,activeSwordAmount, treeAmount, superTreeAmount, spearAmount, netAmount)
67                   elif doneTurn.collidepoint(event.pos):
68                       #disasters execpt for bear attack
69                       #for every 2 people that doesn't have a blanket one will die but if everyone has a blanket it gets removed
70                       if peopleAmount//2<=blanketAmount and disasters.count("cold night")>0:
71                           #remove cold night form list of disaster
72                           disasters.remove("cold night")
73                       else:
74                           #lose people
75                           peopleAmount=disaster.coldNight(peopleAmount, blanketAmount, disasters)
76   
77                       #do the volcano disaster
78                       peopleAmount, disasters=disaster.volcano(peopleAmount, boatAmount, disasters)
79   
80                       #clear disasters that have already been fulfilled
81                       if disasters.count("bear attack")>0:
82                           disasters.remove("bear attack")
83                       if disasters.count("volcano")>0:
84                           disasters.remove("volcano")
85   
86                       #feed people
87                       peopleAmount,coconutAmount=toolBox.feed(peopleAmount,coconutAmount)
88                       #generate coconuts
89                       coconutAmount, disasters=toolBox.generateCoconuts(coconutAmount,treeAmount,superTreeAmount,spearAmount,netAmount,turn, disasters)
90                       #add disasters
91                       disasters.append(disaster.disaster())
92                       turn+=1
93                       #bear attack results
94                       peopleAmount=disaster.bearAttack(peopleAmount,unactiveSwordAmount+activeSwordAmount, disasters)
95   
96                       #player loses
97                       if peopleAmount<=0:
98                           toolBox.loseScreen()
99                           return
100                      #player wins
101                      elif peopleAmount>=settings.amountOfPeopleToWin:
102                           toolBox.winScreen()
103                           return
104  
105                  #display resource window
106                  elif resourceButton.collidepoint(event.pos):
107                      toolBox.resourcesForWindow(peopleAmount, coconutAmount, boatAmount, blanketAmount, medicineAmount, activeSwordAmount, unactiveSwordAmount, treeAmount, superTreeAmount, spearAmount, netAmount)
108                  #start trade
109                  elif tradeButton.collidepoint(event.pos):
110                      peopleAmount,coconutAmount,boatAmount,medicineAmount,blanketAmount,activeSwordAmount,treeAmount,superTreeAmount,spearAmount,netAmount=toolBox.tradeButtons(peopleAmount,coconutAmount,boatAmount,medicineAmount,blanketAmount,unactiveSwordAmount+activeSwordAmount,treeAmount,superTreeAmount,spearAmount,netAmount)
111  
112                  #quit game
113                  elif quitButton.collidepoint(event.pos):
114                      return
115  
116          pygame.display.flip()
