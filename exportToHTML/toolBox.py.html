16       elif whenDividingIsItAWholeNum(turn,settings.superTreeTurnsForYeild) and disasters.count("tree disease")<1:
17           coconutAmount+=settings.superTreeYeild*superTreeAmount
18   
19       # get tree yeild if correct turn and no spear disease
20       elif whenDividingIsItAWholeNum(turn,settings.spearTurnsForYeild) and disasters.count("animal disease")<1:
21           coconutAmount+=settings.spearYeild*spearAmount
22   
23       #get tree yeild if correct turn and no net disease
24       elif whenDividingIsItAWholeNum(turn,settings.netTurnsForYeild) and disasters.count("fish disease")<1:
25           coconutAmount+=settings.netYeild*netAmount

27       #remove disease disasters
28       disaster.removeDisaster("tree disease",disasters)

293  def removeAllFromArray(array, elementToBeRemoved):
294      tempArray=array.remove(elementToBeRemoved)
